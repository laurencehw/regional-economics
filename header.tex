% header.tex — Custom LaTeX preamble for book PDF build

% --- Font fix: Cambria's fi/fl ligatures have broken Unicode cmap entries ---
% in XeLaTeX, causing "fi" to render/extract as U+03D0 (ϐ). Disabling
% common ligatures avoids this; fi/fl as separate glyphs are visually fine.
% TeX ligatures (-- → en-dash, --- → em-dash, `` → quotes) are kept.
\defaultfontfeatures{Ligatures=TeX}
\setmainfont{Cambria}[Ligatures={TeX,NoCommon}]
\setsansfont{Calibri}[Ligatures={TeX,NoCommon}]
